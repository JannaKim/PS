    c = [ max(c[0], y), min(c[1], x) ]
    d = [ max(d[0], y), max(d[1], x) ]

    cnt = 1
    for dy, dx in zip(Dy, Dx):

            vis[ny][nx] = True ## 이거 잘 먹히겠지?
            cnt += dfs(ny, nx, color, grid, vis)
