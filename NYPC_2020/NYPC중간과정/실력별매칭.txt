# input
n = int(input())
s, k = [int(x) for x in input().split()]
score = [int(x) for x in input().split()]
queue = []
for i in range(n):
    queue.append((i+1, score[i]))

def keyfunction(x):
    return abs(x-s) + (0.5 if x > s else 0)

queue.sort(key = lambda x : keyfunction(x[1]))
for i in range(k):
    print(queue[i][0], end = ' ')
